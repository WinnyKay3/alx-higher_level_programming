First I Import the class which is MyList to make it available to this file
	
	>>> MyList = __import__('1-my_list').MyList
	>>> my_list = MyList()
	>>> print(my_list)
	[]

	>>> my_list.sort()
	>>> my_list.print_sorted()
	[]
	>>> my_list[:] = [-4, -2345, -6, -8]
	>>> my_list.print_sorted()
	[-2345, -8, -6, -4]

	>>> my_list[:] = [1, 4, 2, 3, 5]
	>>> my_list.print_sorted()
	[1, 2, 3, 4, 5]

	>>> my_list = MyList()
	>>> my_list[:] = [1, 2, 3, 4, 5]
	>>> my_list.append(6)
	>>> my_list.print_sorted()
	[1, 2, 3, 4, 5, 6]
	
	>>> my_list = []
	>>> my_list.append(1)
	>>> my_list.append(2)
	>>> my_list.append(4)
	>>> my_list.append(5)
	>>> print(my_list)
	[1, 2, 4, 5]
